Model Registry, Inference, and Monitoring

Goal: log model versions, run governed inference, and monitor model behavior over time.

The lab turns trained models into governed Snowflake model versions, then uses those versions for inference, explainability, tags, stored-procedure scoring, and model monitors.

The Lifecycle

Log
Registry.log_model creates a named model version with sample input data.
Score
mv.run(..., function_name="predict") applies the model to Snowpark data.
Promote
Default versions and tags identify the production candidate.
Monitor
CREATE MODEL MONITOR tracks performance and drift against baseline data.

Read This Code

model_registry = Registry(
    session=session,
    database_name=DB,
    schema_name=SCHEMA,
    options={"enable_monitoring": True},
)

mv_base = model_registry.log_model(
    model_name=model_name,
    model=xgb_base,
    version_name="XGB_BASE",
    sample_input_data=train.drop(["TIMESTAMP", "LOAN_ID", "MORTGAGERESPONSE"]).limit(100),
    target_platforms=["WAREHOUSE", "SNOWPARK_CONTAINER_SERVICES"],
    options={"enable_explainability": True},
)

reg_preds = (
    mv_base.run(test, function_name="predict")
    .rename(col('"output_feature_0"'), "MORTGAGE_PREDICTION")
)

model_registry.get_model(model_name).default = "XGB_Optimized"

Monitor Shape

CREATE OR REPLACE MODEL MONITOR MORTGAGE_LENDING_OPTIMIZED_MODEL_MONITOR
WITH
    MODEL = MORTGAGE_LENDING_MLOPS_0
    VERSION = XGB_Optimized
    FUNCTION = predict
    SOURCE = DEMO_MORTGAGE_LENDING_TEST_0
    BASELINE = DEMO_MORTGAGE_LENDING_TRAIN_0
    TIMESTAMP_COLUMN = TIMESTAMP
    PREDICTION_CLASS_COLUMNS = (XGB_OPTIMIZED_PREDICTION)
    ACTUAL_CLASS_COLUMNS = (MORTGAGERESPONSE)
    ID_COLUMNS = (LOAN_ID)
    WAREHOUSE = E2E_SNOW_MLOPS_WH
    REFRESH_INTERVAL = '12 hours'
    AGGREGATION_WINDOW = '1 day';
