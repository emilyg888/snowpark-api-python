Snowpark DataFrame API Basics

A compact reference for the first Snowpark DataFrame patterns.

Mental Model

DataFrame
A lazy relational query plan.
Transformation
Returns a new DataFrame plan.
Action
Runs SQL in Snowflake and returns results.

Create

df = session.table("MY_TABLE")
df = session.sql("select * from MY_TABLE")
df = session.create_dataframe([[1, "A"], [2, "B"]], schema=["id", "name"])

Transform

from snowflake.snowpark.functions import col, lit, upper, sum as sum_

result = (
    session.table("ORDERS")
    .filter(col("STATUS") == lit("SHIPPED"))
    .select(
        col("CUSTOMER_ID"),
        col("AMOUNT"),
        upper(col("REGION")).as_("REGION")
    )
    .group_by("CUSTOMER_ID", "REGION")
    .agg(sum_("AMOUNT").as_("TOTAL_AMOUNT"))
    .sort(col("TOTAL_AMOUNT").desc())
)

Join

orders = session.table("ORDERS")
customers = session.table("CUSTOMERS")

joined = (
    orders
    .join(customers, orders["CUSTOMER_ID"] == customers["ID"])
    .select(
        orders["ORDER_ID"],
        customers["NAME"].as_("CUSTOMER_NAME"),
        orders["AMOUNT"]
    )
)
